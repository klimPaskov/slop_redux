slopx_ireland_update_mechanics_modifier = {
	slopx_ireland_clamp_mechanics = yes

	set_variable = { ireland_dynamic_political_power_factor = -0.04 }
	set_variable = { ireland_dynamic_stability_factor = -0.05 }
	set_variable = { ireland_dynamic_war_support_factor = 0 }
	set_variable = { ireland_dynamic_army_defence_factor = -0.04 }
	set_variable = { ireland_dynamic_factory_output_factor = -0.03 }
	set_variable = { ireland_dynamic_consumer_goods_factor = 0.04 }

	if = {
		limit = { check_variable = { ireland_constitutional_authority > 49 } }
		set_variable = { ireland_dynamic_political_power_factor = 0 }
		set_variable = { ireland_dynamic_stability_factor = 0 }
	}
	if = {
		limit = { check_variable = { ireland_constitutional_authority > 69 } }
		set_variable = { ireland_dynamic_political_power_factor = 0.04 }
		set_variable = { ireland_dynamic_stability_factor = 0.03 }
	}
	if = {
		limit = { check_variable = { ireland_constitutional_authority > 84 } }
		set_variable = { ireland_dynamic_political_power_factor = 0.06 }
		set_variable = { ireland_dynamic_stability_factor = 0.05 }
	}
	if = {
		limit = { check_variable = { ireland_emergency_preparedness > 24 } }
		set_variable = { ireland_dynamic_army_defence_factor = 0 }
	}
	if = {
		limit = { check_variable = { ireland_emergency_preparedness > 54 } }
		set_variable = { ireland_dynamic_war_support_factor = 0.04 }
		set_variable = { ireland_dynamic_army_defence_factor = 0.03 }
	}
	if = {
		limit = { check_variable = { ireland_emergency_preparedness > 74 } }
		set_variable = { ireland_dynamic_war_support_factor = 0.07 }
		set_variable = { ireland_dynamic_army_defence_factor = 0.06 }
	}
	if = {
		limit = { check_variable = { ireland_foreign_access_pressure < 45 } }
		set_variable = { ireland_dynamic_factory_output_factor = 0 }
		set_variable = { ireland_dynamic_consumer_goods_factor = 0.02 }
	}
	if = {
		limit = { check_variable = { ireland_foreign_access_pressure < 25 } }
		set_variable = { ireland_dynamic_factory_output_factor = 0.03 }
		set_variable = { ireland_dynamic_consumer_goods_factor = 0 }
	}
	if = {
		limit = { check_variable = { ireland_partition_pressure > 74 } }
		add_to_variable = { ireland_dynamic_stability_factor = -0.04 }
	}
	force_update_dynamic_modifier = yes
}
