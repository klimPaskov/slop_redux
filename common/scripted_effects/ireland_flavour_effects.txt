@ireland_flavour_global_cooldown = 60
@ireland_flavour_war_cooldown = 45

# Ireland flavour event effects.
# Shared pulse, cooldown, task, and family outcome helpers for the row-level flavour event layer.

slopx_ireland_flavour_pulse = {
	if = {
		limit = {
			original_tag = IRE
			has_country_flag = ireland_mechanics_initialized
			NOT = { has_country_flag = ireland_flavour_global_cooldown }
		}
		country_event = { id = slopx_ireland_flavour.1 days = 1 }
	}
}

slopx_ireland_flavour_set_cooldown = {
	if = {
		limit = { has_war = yes }
		set_variable = { ireland_flavour_cooldown_days = @ireland_flavour_war_cooldown }
	}
	else = {
		set_variable = { ireland_flavour_cooldown_days = @ireland_flavour_global_cooldown }
	}
	set_country_flag = { flag = ireland_flavour_global_cooldown days = var:ireland_flavour_cooldown_days }
	add_to_variable = { ireland_flavour_events_seen = 1 }
}
