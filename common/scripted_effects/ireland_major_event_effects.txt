@ireland_major_event_cooldown = 25
@ireland_major_cleanup_cooldown = 45

# Ireland major-event dispatcher helpers. These are pulsed only from on_daily_IRE.
slopx_ireland_major_event_pulse = {
	if = {
		limit = {
			original_tag = IRE
			has_country_flag = ireland_mechanics_initialized
			NOT = { has_country_flag = ireland_major_cleanup_cooldown }
		}
		slopx_ireland_major_cleanup_set_cooldown = yes
		country_event = { id = slopx_ireland_focus.131 days = 2 }
	}
}

slopx_ireland_major_event_set_cooldown = {
	set_variable = { ireland_major_event_cooldown_days = @ireland_major_event_cooldown }
	set_country_flag = { flag = ireland_major_event_cooldown days = var:ireland_major_event_cooldown_days }
	add_to_variable = { ireland_major_events_seen = 1 }
}

slopx_ireland_major_cleanup_set_cooldown = {
	set_variable = { ireland_major_cleanup_cooldown_days = @ireland_major_cleanup_cooldown }
	set_country_flag = { flag = ireland_major_cleanup_cooldown days = var:ireland_major_cleanup_cooldown_days }
}
