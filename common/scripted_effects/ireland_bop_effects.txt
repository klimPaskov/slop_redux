ire_bop_preserve_value_conservatively_effect = {
	if = {
		limit = {
			power_balance_value = {
				id = IRE_state_authority_bop
				value < @ire_bop_mode_translation_left_extreme
			}
		}
		set_power_balance = {
			id = IRE_state_authority_bop
			set_value = @ire_bop_mode_translation_left_cap
		}
	}
	if = {
		limit = {
			power_balance_value = {
				id = IRE_state_authority_bop
				value > @ire_bop_mode_translation_right_extreme
			}
		}
		set_power_balance = {
			id = IRE_state_authority_bop
			set_value = @ire_bop_mode_translation_right_cap
		}
	}
}

ire_bop_initialize_opening_effect = {
	ire_bop_clear_mode_flags_effect = yes
	set_country_flag = ireland_bop_initialized
	set_country_flag = ire_bop_mode_opening_state
	set_power_balance = {
		id = IRE_state_authority_bop
		left_side = ire_bop_side_constitutional_government
		right_side = ire_bop_side_armed_political_pressure
		set_value = @ire_bop_opening_start
	}
}
