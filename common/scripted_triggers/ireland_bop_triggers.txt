# Ireland state authority BOP helper triggers.
# These triggers assume the parent BOP object defines the range ids documented here.

ire_bop_has_state_authority_bop_trigger = {
	has_power_balance = {
		id = IRE_state_authority_bop
	}
}

ire_bop_has_state_authority = {
	ire_bop_has_state_authority_bop_trigger = yes
}

ire_bop_is_opening_mode_trigger = {
	has_country_flag = ire_bop_mode_opening_state
	ire_bop_has_state_authority_bop_trigger = yes
	is_power_balance_side_active = {
		id = IRE_state_authority_bop
		side = ire_bop_side_constitutional_government
	}
	is_power_balance_side_active = {
		id = IRE_state_authority_bop
		side = ire_bop_side_armed_political_pressure
	}
}
